51fn determine_mtu(dev_cfg: &NetDevCfg) -> u16 {
52	// If VIRTIO_NET_F_MTU is negotiated, "the driver uses mtu as the maximum MTU value"
53	// (VirtIO specification, 5.1.3, "Feature bits")
54	if dev_cfg.features.contains(virtio::net::F::MTU) {
55		dev_cfg.raw.as_ptr().mtu().read().to_ne()
56	} else {
57		// Otherwise, we can just use the MTU we want to use
58		mtu()
59	}
60}

62fn determine_buf_size(dev_cfg: &NetDevCfg) -> u32 {
63	// See Virtio specification v1.1 - 5.1.6.3.1 and 5.1.4.2
64
65	// Our desired minimum buffer size - we want it to be at least the MTU generally
66	let mut min_buf_size = determine_mtu(dev_cfg).into();
67
68	// If VIRTIO_NET_F_MRG_RXBUF is negotiated, each buffer MUST be at least the size of the struct virtio_net_hdr.
69	// We just use MTU in that case, but otherwise...
70	if !dev_cfg.features.contains(virtio::net::F::MRG_RXBUF) {
71		// If [...] are negotiated, the driver SHOULD populate the receive queue(s) with buffers of at least 65562 bytes.
72		if dev_cfg.features.contains(virtio::net::F::GUEST_TSO4)
73			|| dev_cfg.features.contains(virtio::net::F::GUEST_TSO6)
74			|| dev_cfg.features.contains(virtio::net::F::GUEST_UFO)
75		{
76			min_buf_size = u32::max(min_buf_size, 65562 - size_of::<Hdr>() as u32);
77		} else {
78			// Otherwise, the driver SHOULD populate the receive queue(s) with buffers of at least 1526 bytes.
79			min_buf_size = u32::max(min_buf_size, 1526 - size_of::<Hdr>() as u32);
80		}
81	}
82
83	min_buf_size
84}

91impl RxQueues {
92	pub fn new(vqs: Vec<VirtQueue>, dev_cfg: &NetDevCfg) -> Self {
93		Self {
94			vqs,
95			buf_size: determine_buf_size(dev_cfg),
96		}
97	}

99	/// Takes care of handling packets correctly which need some processing after being received.
100	/// This currently include nothing. But in the future it might include among others:
101	/// * Calculating missing checksums
102	/// * Merging receive buffers, by simply checking the poll_queue (if VIRTIO_NET_F_MRG_BUF)
103	fn post_processing(_buffer_tkn: &mut UsedBufferToken) -> Result<(), VirtioNetError> {
104		Ok(())
105	}
106
107	/// Adds a given queue to the underlying vector and populates the queue with RecvBuffers.
108	///
109	/// Queues are all populated according to Virtio specification v1.1. - 5.1.6.3.1
110	fn add(&mut self, mut vq: VirtQueue) {
111		const BUFF_PER_PACKET: u16 = 2;
112		let num_bufs: u16 = u16::from(vq.size()) / BUFF_PER_PACKET;
113		fill_queue(&mut vq, num_bufs, self.buf_size);
114		self.vqs.push(vq);
115	}
116
117	fn get_next(&mut self) -> Option<UsedBufferToken> {
118		self.vqs[0].try_recv().ok()
119	}
120
121	fn enable_notifs(&mut self) {
122		for vq in &mut self.vqs {
123			vq.enable_notifs();
124		}
125	}
126
127	fn disable_notifs(&mut self) {
128		for vq in &mut self.vqs {
129			vq.disable_notifs();
130		}
131	}
132
133	fn has_packet(&self) -> bool {
134		self.vqs.iter().any(|vq| vq.has_used_buffers())
135	}
136}

138fn buffer_token_from_hdr(
139	hdr: Box<MaybeUninit<Hdr>, DeviceAlloc>,
140	buf_size: u32,
141) -> AvailBufferToken {
142	AvailBufferToken::new(SmallVec::new(), {
143		SmallVec::from_buf([
144			BufferElem::Sized(hdr),
145			BufferElem::Vector(Vec::with_capacity_in(
146				buf_size.try_into().unwrap(),
147				DeviceAlloc,
148			)),
149		])
150	})
151	.unwrap()
152}
153
154fn fill_queue(vq: &mut VirtQueue, num_bufs: u16, buf_size: u32) {
155	for _ in 0..num_bufs {
156		let buff_tkn = buffer_token_from_hdr(Box::<Hdr, _>::new_uninit_in(DeviceAlloc), buf_size);
157
158		// BufferTokens are directly provided to the queue
159		// TransferTokens are directly dispatched
160		// Transfers will be awaited at the queue
161		if let Err(err) = vq.dispatch(buff_tkn, false, BufferType::Direct) {
162			error!("{err:#?}");
163			break;
164		}
165	}
166}

168/// Structure which handles transmission of packets and delegation
169/// to the respective queue structures.
170pub struct TxQueues {
171	vqs: Vec<VirtQueue>,
172	/// Indicates, whether the Driver/Device are using multiple
173	/// queues for communication.
174	buf_size: u32,
175}
176
177impl TxQueues {
178	pub fn new(vqs: Vec<VirtQueue>, dev_cfg: &NetDevCfg) -> Self {
179		Self {
180			vqs,
181			buf_size: determine_buf_size(dev_cfg),
182		}
183	}
184
185	#[allow(dead_code)]
186	fn enable_notifs(&mut self) {
187		for vq in &mut self.vqs {
188			vq.enable_notifs();
189		}
190	}
191
192	#[allow(dead_code)]
193	fn disable_notifs(&mut self) {
194		for vq in &mut self.vqs {
195			vq.disable_notifs();
196		}
197	}
198
199	fn poll(&mut self) {
200		for vq in &mut self.vqs {
201			// We don't do anything with the buffers but we need to receive them for the
202			// ring slots to be emptied and the memory from the previous transfers to be freed.
203			while vq.try_recv().is_ok() {}
204		}
205	}
206
207	fn add(&mut self, vq: VirtQueue) {
208		// Currently we are doing nothing with the additional queues. They are inactive and might be used in the
209		// future
210		self.vqs.push(vq);
211	}
212}

265
266	/// Provides smoltcp a slice to copy the IP packet and transfer the packet
267	/// to the send queue.
268	fn send_packet<R, F>(&mut self, len: usize, f: F) -> R
269	where
270		F: FnOnce(&mut [u8]) -> R,
271	{
272		// We need to poll to get the queue to remove elements from the table and make space for
273		// what we are about to add
274		self.inner.send_vqs.poll();
275
276		assert!(len < usize::try_from(self.inner.send_vqs.buf_size).unwrap());
277		let mut packet = Vec::with_capacity_in(len, DeviceAlloc);
278		let result = unsafe {
279			let result = f(packet.spare_capacity_mut().assume_init_mut());
280			packet.set_len(len);
281			result
282		};
283
284		let mut header = Box::new_in(<Hdr as Default>::default(), DeviceAlloc);
285		// If a checksum isn't necessary, we have inform the host within the header
286		// see Virtio specification 5.1.6.2
287		if !self.checksums.tcp.tx() || !self.checksums.udp.tx() {
288			header.flags = HdrF::NEEDS_CSUM;
289			let ethernet_frame: smoltcp::wire::EthernetFrame<&[u8]> =
290				EthernetFrame::new_unchecked(&packet);
291			let packet_header_len: u16;
292			let protocol;
293			match ethernet_frame.ethertype() {
294				smoltcp::wire::EthernetProtocol::Ipv4 => {
295					let packet = Ipv4Packet::new_unchecked(ethernet_frame.payload());
296					packet_header_len = packet.header_len().into();
297					protocol = Some(packet.next_header());
298				}
299				smoltcp::wire::EthernetProtocol::Ipv6 => {
300					let packet = Ipv6Packet::new_unchecked(ethernet_frame.payload());
301					packet_header_len = packet.header_len().try_into().unwrap();
302					protocol = Some(packet.next_header());
303				}
304				_ => {
305					packet_header_len = 0;
306					protocol = None;
307				}
308			}
309			header.csum_start =
310				(u16::try_from(ETHERNET_HEADER_LEN).unwrap() + packet_header_len).into();
311			header.csum_offset = match protocol {
312				Some(smoltcp::wire::IpProtocol::Tcp) => 16,
313				Some(smoltcp::wire::IpProtocol::Udp) => 6,
314				_ => 0,
315			}
316			.into();
317		}
318
319		let buff_tkn = AvailBufferToken::new(
320			SmallVec::from_buf([BufferElem::Sized(header), BufferElem::Vector(packet)]),
321			SmallVec::new(),
322		)
323		.unwrap();
324
325		self.inner.send_vqs.vqs[0]
326			.dispatch(buff_tkn, false, BufferType::Direct)
327			.unwrap();
328
329		result
330	}
331
332	fn receive_packet(&mut self) -> Option<(RxToken, TxToken)> {
333		let (first_header, first_packet) = self.receive_single_packet()?;
334
335		// According to VIRTIO spec v1.2 sec. 5.1.6.3.2, "num_buffers will always be 1 if VIRTIO_NET_F_MRG_RXBUF is not negotiated."
336		// Unfortunately, NVIDIA MLX5 does not comply with this requirement and we have to manually set the value to the correct one.
337		let num_buffers = if self.dev_cfg.features.contains(virtio::net::F::MRG_RXBUF) {
338			first_header.num_buffers.to_ne()
339		} else {
340			1
341		};
342
343		let mut combined_packets = first_packet;
344
345		let first_tkn = buffer_token_from_hdr(
346			// SAFETY: Box<T> -> Box<MaybeUninit<T>> is sound
347			unsafe {
348				transmute::<Box<Hdr, DeviceAlloc>, Box<MaybeUninit<Hdr>, DeviceAlloc>>(first_header)
349			},
350			self.inner.recv_vqs.buf_size,
351		);
352		self.inner.recv_vqs.vqs[0]
353			.dispatch(first_tkn, false, BufferType::Direct)
354			.unwrap();
355
356		for _ in 1..num_buffers {
357			let (header, packet) = self.receive_single_packet()?;
358			combined_packets.extend_from_slice(&packet);
359
360			let tkn = buffer_token_from_hdr(
361				// SAFETY: Box<T> -> Box<MaybeUninit<T>> is sound
362				unsafe {
363					transmute::<Box<Hdr, DeviceAlloc>, Box<MaybeUninit<Hdr>, DeviceAlloc>>(header)
364				},
365				self.inner.recv_vqs.buf_size,
366			);
367			self.inner.recv_vqs.vqs[0]
368				.dispatch(tkn, false, BufferType::Direct)
369				.unwrap();
370		}
371
372		Some((RxToken::new(combined_packets), TxToken::new()))
373	}

375	fn set_polling_mode(&mut self, value: bool) {
376		if value {
377			self.disable_interrupts();
378		} else {
379			self.enable_interrupts();
380		}

383	fn handle_interrupt(&mut self) {
384		let status = self.isr_stat.is_queue_interrupt();
385
386		#[cfg(not(feature = "pci"))]
387		if status.contains(virtio::mmio::InterruptStatus::CONFIGURATION_CHANGE_NOTIFICATION) {
388			info!("Configuration changes are not possible! Aborting");
389			todo!("Implement possibility to change config on the fly...")
390		}
391
392		#[cfg(feature = "pci")]
393		if status.contains(virtio::pci::IsrStatus::DEVICE_CONFIGURATION_INTERRUPT) {
394			info!("Configuration changes are not possible! Aborting");
395			todo!("Implement possibility to change config on the fly...")
396		}
397
398		self.isr_stat.acknowledge();
399	}

491	fn receive_single_packet(&mut self) -> Option<(Box<Hdr, DeviceAlloc>, Vec<u8, DeviceAlloc>)> {
492		let mut buffer_tkn = self.inner.recv_vqs.get_next()?;
493		RxQueues::post_processing(&mut buffer_tkn)

503impl VirtioNetDriver<Uninit> {
504	/// Initializes the device in adherence to specification. Returns Some(VirtioNetError)
505	/// upon failure and None in case everything worked as expected.
506	///
507	/// See Virtio specification v1.1. - 3.1.1.
508	///                      and v1.1. - 5.1.5
509	pub fn init_dev(mut self) -> Result<VirtioNetDriver<Init>, VirtioNetError> {
510		// Reset
511		self.com_cfg.reset_dev();
512
513		// Indicate device, that OS noticed it
514		self.com_cfg.ack_dev();
515
516		// Indicate device, that driver is able to handle it
517		self.com_cfg.set_drv();
518
519		let minimal_features = virtio::net::F::VERSION_1 | virtio::net::F::MAC;
520
521		// If wanted, push new features into feats here:
522		let mut features = minimal_features
523			// Indirect descriptors can be used
524			| virtio::net::F::INDIRECT_DESC
525			// Packed Vq can be used
526			| virtio::net::F::RING_PACKED
527			| virtio::net::F::NOTIFICATION_DATA
528			// Host should avoid the creation of checksums
529			| virtio::net::F::CSUM
530			// Guest avoids the creation of checksums
531			| virtio::net::F::GUEST_CSUM
532			// MTU setting can be used
533			| virtio::net::F::MTU
534			// Driver can merge receive buffers
535			| virtio::net::F::MRG_RXBUF
536			// the link status can be announced
537			| virtio::net::F::STATUS
538			// control queue support
539			| virtio::net::F::CTRL_VQ
540			// Multiqueue support
541			| virtio::net::F::MQ;
542
543		// Currently the driver does NOT support the features below.
544		// In order to provide functionality for these, the driver
545		// needs to take care of calculating checksum in
546		// RxQueues.post_processing()
547		// | virtio::net::F::GUEST_TSO4
548		// | virtio::net::F::GUEST_TSO6
549
550		// Negotiate features with device. Automatically reduces selected feats in order to meet device capabilities.
551		// Aborts in case incompatible features are selected by the driver or the device does not support min_feat_set.
552		match self.negotiate_features(features) {
553			Ok(()) => info!(
554				"Driver found a subset of features for virtio device {:x}. Features are: {features:?}",
555				self.dev_cfg.dev_id
556			),
557			Err(vnet_err) => {
558				match vnet_err {
559					VirtioNetError::FeatureRequirementsNotMet(features) => {
560						error!(
561							"Network drivers feature set {features:?} does not satisfy rules in section 5.1.3.1 of specification v1.1. Aborting!"
562						);
563						return Err(vnet_err);
564					}
565					VirtioNetError::IncompatibleFeatureSets(drv_feats, dev_feats) => {
566						// Create a new matching feature set for device and driver if the minimal set is met!
567						if !dev_feats.contains(minimal_features) {
568							error!(
569								"Device features set, does not satisfy minimal features needed. Aborting!"
570							);
571							return Err(VirtioNetError::FailFeatureNeg(self.dev_cfg.dev_id));
572						}
573
574						let common_features = drv_feats & dev_feats;
575						if common_features.is_empty() {
576							error!(
577								"Feature negotiation failed with minimal feature set. Aborting!"
578							);
579							return Err(VirtioNetError::FailFeatureNeg(self.dev_cfg.dev_id));
580						}
581						features = common_features;
582
583						match self.negotiate_features(features) {
584							Ok(()) => info!(
585								"Driver found a subset of features for virtio device {:x}. Features are: {features:?}",
586								self.dev_cfg.dev_id
587							),
588							Err(vnet_err) => match vnet_err {
589								VirtioNetError::FeatureRequirementsNotMet(features) => {
590									error!(
591										"Network device offers a feature set {features:?} when used completely does not satisfy rules in section 5.1.3.1 of specification v1.1. Aborting!"
592									);
593									return Err(vnet_err);
594								}
595								_ => {
596									error!(
597										"Feature Set after reduction still not usable. Set: {features:?}. Aborting!"
598									);
599									return Err(vnet_err);
600								}
601							},
602						}
603					}
604					VirtioNetError::FailFeatureNeg(_) => {
605						error!(
606							"Wanted set of features is NOT supported by device. Set: {features:?}"
607						);
608						return Err(vnet_err);
609					}
610					#[cfg(feature = "pci")]
611					VirtioNetError::NoDevCfg(_) => {
612						error!("No device config found.");
613						return Err(vnet_err);
614					}
615				}
616			}
617		}
618
619		// Indicates the device, that the current feature set is final for the driver
620		// and will not be changed.
621		self.com_cfg.features_ok();
622
623		// Checks if the device has accepted final set. This finishes feature negotiation.
624		if self.com_cfg.check_features() {
625			info!(
626				"Features have been negotiated between virtio network device {:x} and driver.",
627				self.dev_cfg.dev_id
628			);
629			// Set feature set in device config fur future use.
630			self.dev_cfg.features = features;
631		} else {
632			return Err(VirtioNetError::FailFeatureNeg(self.dev_cfg.dev_id));
633		}
634
635		let mut inner = Init {
636			mtu: determine_mtu(&self.dev_cfg),
637			ctrl_vq: None,
638			recv_vqs: RxQueues::new(Vec::new(), &self.dev_cfg),
639			send_vqs: TxQueues::new(Vec::new(), &self.dev_cfg),
640		};
641
642		debug!("Using RX buffer size of {}", inner.recv_vqs.buf_size);
643
644		self.dev_spec_init(&mut inner)?;
645		info!(
646			"Device specific initialization for Virtio network device {:x} finished",
647			self.dev_cfg.dev_id
648		);
649
650		// At this point the device is "live"
651		self.com_cfg.drv_ok();
652
653		if self.dev_cfg.features.contains(virtio::net::F::CSUM)
654			&& self.dev_cfg.features.contains(virtio::net::F::GUEST_CSUM)
655		{
656			self.checksums.udp = Checksum::None;
657			self.checksums.tcp = Checksum::None;
658		} else if self.dev_cfg.features.contains(virtio::net::F::CSUM) {
659			self.checksums.udp = Checksum::Rx;
660			self.checksums.tcp = Checksum::Rx;
661		} else if self.dev_cfg.features.contains(virtio::net::F::GUEST_CSUM) {
662			self.checksums.udp = Checksum::Tx;
663			self.checksums.tcp = Checksum::Tx;
664		}
665		debug!("{:?}", self.checksums);
666
667		Ok(VirtioNetDriver {
668			dev_cfg: self.dev_cfg,
669			com_cfg: self.com_cfg,
670			isr_stat: self.isr_stat,
671			notif_cfg: self.notif_cfg,
672			inner,
673			num_vqs: self.num_vqs,
674			irq: self.irq,
675			checksums: self.checksums,
676		})
677	}
678
679	/// Negotiates a subset of features, understood and wanted by both the OS
680	/// and the device.
681	fn negotiate_features(
682		&mut self,
683		driver_features: virtio::net::F,
684	) -> Result<(), VirtioNetError> {
685		let device_features = virtio::net::F::from(self.com_cfg.dev_features());
686
687		if device_features.requirements_satisfied() {
688			info!("Feature set wanted by network driver are in conformance with specification.");
689		} else {
690			return Err(VirtioNetError::FeatureRequirementsNotMet(device_features));
691		}
692
693		if device_features.contains(driver_features) {
694			// If device supports subset of features write feature set to common config
695			self.com_cfg.set_drv_features(driver_features.into());
696			Ok(())
697		} else {
698			Err(VirtioNetError::IncompatibleFeatureSets(
699				driver_features,
700				device_features,
701			))
702		}
703	}
704
705	/// Device Specific initialization according to Virtio specifictation v1.1. - 5.1.5
706	fn dev_spec_init(&mut self, inner: &mut Init) -> Result<(), VirtioNetError> {
707		self.virtqueue_init(inner)?;
708		info!("Network driver successfully initialized virtqueues.");
709
710		// Add a control if feature is negotiated
711		if self.dev_cfg.features.contains(virtio::net::F::CTRL_VQ) {
712			let mut ctrl_vq = if self.dev_cfg.features.contains(virtio::net::F::RING_PACKED) {
713				VirtQueue::Packed(
714					PackedVq::new(
715						&mut self.com_cfg,
716						&self.notif_cfg,
717						VqSize::from(VIRTIO_MAX_QUEUE_SIZE),
718						VqIndex::from(self.num_vqs),
719						self.dev_cfg.features.into(),
720					)
721					.unwrap(),
722				)
723			} else {
724				VirtQueue::Split(
725					SplitVq::new(
726						&mut self.com_cfg,
727						&self.notif_cfg,
728						VqSize::from(VIRTIO_MAX_QUEUE_SIZE),
729						VqIndex::from(self.num_vqs),
730						self.dev_cfg.features.into(),
731					)
732					.unwrap(),
733				)
734			};
735
736			ctrl_vq.enable_notifs();
737			inner.ctrl_vq = Some(ctrl_vq);
738		}
739
740		Ok(())
741	}
742
743	/// Initialize virtqueues via the queue interface and populates receiving queues
744	fn virtqueue_init(&mut self, inner: &mut Init) -> Result<(), VirtioNetError> {
745		// We are assuming here, that the device single source of truth is the
746		// device specific configuration. Hence we do NOT check if
747		//
748		// max_virtqueue_pairs + 1 < num_queues
749		//
750		// - the plus 1 is due to the possibility of an existing control queue
751		// - the num_queues is found in the ComCfg struct of the device and defines the maximal number
752		// of supported queues.
753		if self.dev_cfg.features.contains(virtio::net::F::MQ) {
754			if self
755				.dev_cfg
756				.raw
757				.as_ptr()
758				.max_virtqueue_pairs()
759				.read()
760				.to_ne() * 2 >= MAX_NUM_VQ
761			{
762				self.num_vqs = MAX_NUM_VQ;
763			} else {
764				self.num_vqs = self
765					.dev_cfg
766					.raw
767					.as_ptr()
768					.max_virtqueue_pairs()
769					.read()
770					.to_ne() * 2;
771			}
772		} else {
773			// Minimal number of virtqueues defined in the standard v1.1. - 5.1.5 Step 1
774			self.num_vqs = 2;
775		}
776
777		// The loop is running from 0 to num_vqs and the indexes are provided to the VqIndex::from function in this way
778		// in order to allow the indexes of the queues to be in a form of:
779		//
780		// index i for receive queue
781		// index i+1 for send queue
782		//
783		// as it is wanted by the network network device.
784		// see Virtio specification v1.1. - 5.1.2
785		// Assure that we have always an even number of queues (i.e. pairs of queues).
786		assert_eq!(self.num_vqs % 2, 0);
787
788		for i in 0..(self.num_vqs / 2) {
789			if self.dev_cfg.features.contains(virtio::net::F::RING_PACKED) {
790				let mut vq = PackedVq::new(
791					&mut self.com_cfg,
792					&self.notif_cfg,
793					VqSize::from(VIRTIO_MAX_QUEUE_SIZE),
794					VqIndex::from(2 * i),
795					self.dev_cfg.features.into(),
796				)
797				.unwrap();
798				// Interrupt for receiving packets is wanted
799				vq.enable_notifs();
800
801				inner.recv_vqs.add(VirtQueue::Packed(vq));
802
803				let mut vq = PackedVq::new(
804					&mut self.com_cfg,
805					&self.notif_cfg,
806					VqSize::from(VIRTIO_MAX_QUEUE_SIZE),
807					VqIndex::from(2 * i + 1),
808					self.dev_cfg.features.into(),
809				)
810				.unwrap();
811				// Interrupt for communicating that a sended packet left, is not needed
812				vq.disable_notifs();
813
814				inner.send_vqs.add(VirtQueue::Packed(vq));
815			} else {
816				let mut vq = SplitVq::new(
817					&mut self.com_cfg,
818					&self.notif_cfg,
819					VqSize::from(VIRTIO_MAX_QUEUE_SIZE),
820					VqIndex::from(2 * i),
821					self.dev_cfg.features.into(),
822				)
823				.unwrap();
824				// Interrupt for receiving packets is wanted
825				vq.enable_notifs();
826
827				inner.recv_vqs.add(VirtQueue::Split(vq));
828
829				let mut vq = SplitVq::new(
830					&mut self.com_cfg,
831					&self.notif_cfg,
832					VqSize::from(VIRTIO_MAX_QUEUE_SIZE),
833					VqIndex::from(2 * i + 1),
834					self.dev_cfg.features.into(),
835				)
836				.unwrap();
837				// Interrupt for communicating that a sended packet left, is not needed
838				vq.disable_notifs();
839
840				inner.send_vqs.add(VirtQueue::Split(vq));
841			}
842		}
843
844		Ok(())
845	}
846}
