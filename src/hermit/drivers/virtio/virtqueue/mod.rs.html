90/// The Virtq trait unifies access to the two different Virtqueue types
91/// [packed::PackedVq] and [split::SplitVq].
92///
93/// The trait provides a common interface for both types. Which in some case
94/// might not provide the complete feature set of each queue. Drivers who
95/// do need these features should refrain from providing support for both
96/// Virtqueue types and use the structs directly instead.
97#[enum_dispatch]
98pub trait Virtq: Send {
99	/// The `notif` parameter indicates if the driver wants to have a notification for this specific
100	/// transfer. This is only for performance optimization. As it is NOT ensured, that the device sees the
101	/// updated notification flags before finishing transfers!
102	fn dispatch(
103		&mut self,
104		tkn: AvailBufferToken,
105		notif: bool,
106		buffer_type: BufferType,
107	) -> Result<(), VirtqError>;
108
109	/// Dispatches the provided TransferToken to the respective queue and does
110	/// return when, the queue finished the transfer.
111	///
112	/// The returned [UsedBufferToken] can be copied from
113	/// or return the underlying buffers.
114	///
115	/// **INFO:**
116	/// Currently this function is constantly polling the queue while keeping the notifications disabled.
117	/// Upon finish notifications are enabled again.
118	fn dispatch_blocking(
119		&mut self,
120		tkn: AvailBufferToken,
121		buffer_type: BufferType,
122	) -> Result<UsedBufferToken, VirtqError> {
123		self.dispatch(tkn, false, buffer_type)?;
124
125		self.disable_notifs();
126
127		let result: UsedBufferToken;
128		// Keep Spinning until the receive queue is filled
129		loop {
130			// TODO: normally, we should check if the used buffer in question is the one
131			// we just made available. However, this shouldn't be a problem as the queue this
132			// function is called on makes use of this blocking dispatch function exclusively
133			// and thus dispatches cannot be interleaved.
134			if let Ok(buffer_tkn) = self.try_recv() {
135				result = buffer_tkn;
136				break;
137			}
138		}
139
140		self.enable_notifs();
141
142		Ok(result)
143	}
144
145	/// Enables interrupts for this virtqueue upon receiving a transfer
146	fn enable_notifs(&mut self);
147
148	/// Disables interrupts for this virtqueue upon receiving a transfer
149	fn disable_notifs(&mut self);
150
151	/// Checks if new used descriptors have been written by the device.
152	/// This activates the queue and polls the descriptor ring of the queue.
153	fn try_recv(&mut self) -> Result<UsedBufferToken, VirtqError>;
154
155	/// Dispatches a batch of [AvailBufferToken]s. The buffers are provided to the queue in
156	/// sequence. After the last buffer has been written, the queue marks the first buffer as available and triggers
157	/// a device notification if wanted by the device.
158	///
159	/// The `notif` parameter indicates if the driver wants to have a notification for this specific
160	/// transfer. This is only for performance optimization. As it is NOT ensured, that the device sees the
161	/// updated notification flags before finishing transfers!
162	fn dispatch_batch(
163		&mut self,
164		tkns: Vec<(AvailBufferToken, BufferType)>,
165		notif: bool,
166	) -> Result<(), VirtqError>;
167
168	/// Dispatches a batch of [AvailBufferToken]s. The tokens will be placed in to the `await_queue`
169	/// upon finish.
170	///
171	/// The `notif` parameter indicates if the driver wants to have a notification for this specific
172	/// transfer. This is only for performance optimization. As it is NOT ensured, that the device sees the
173	/// updated notification flags before finishing transfers!
174	///
175	/// The buffers are provided to the queue in
176	/// sequence. After the last buffer has been written, the queue marks the first buffer as available and triggers
177	/// a device notification if wanted by the device.
178	///
179	/// Tokens to get a reference to the provided await_queue, where they will be placed upon finish.
180	fn dispatch_batch_await(
181		&mut self,
182		tkns: Vec<(AvailBufferToken, BufferType)>,
183		notif: bool,
184	) -> Result<(), VirtqError>;
185
186	/// Returns the size of a Virtqueue. This represents the overall size and not the capacity the
187	/// queue currently has for new descriptors.
188	fn size(&self) -> VqSize;
189
190	// Returns the index (ID) of a Virtqueue.
191	fn index(&self) -> VqIndex;
192
193	fn has_used_buffers(&self) -> bool;
194}

196/// These methods are an implementation detail and are meant only for consumption by the default method
197/// implementations in [Virtq].
198trait VirtqPrivate {
199	type Descriptor: VirtqDescriptor;
200
201	fn create_indirect_ctrl(
202		buffer_tkn: &AvailBufferToken,
203	) -> Result<Box<[Self::Descriptor]>, VirtqError>;
204
205	fn indirect_desc(table: &[Self::Descriptor]) -> Self::Descriptor {
206		let addr = table.as_ptr().expose_provenance();
207		Self::Descriptor::incomplete_desc(
208			u64::try_from(addr).unwrap().into(),
209			(mem::size_of_val(table) as u32).into(),
210			virtq::DescF::INDIRECT,
211		)
212	}
213
214	/// Consumes the [AvailBufferToken] and returns a [TransferToken], that can be used to actually start the transfer.
215	///
216	/// After this call, the buffers are no longer writable.
217	fn transfer_token_from_buffer_token(
218		buff_tkn: AvailBufferToken,
219		buffer_type: BufferType,
220	) -> TransferToken<Self::Descriptor> {
221		let ctrl_desc = match buffer_type {
222			BufferType::Direct => None,
223			BufferType::Indirect => Some(Self::create_indirect_ctrl(&buff_tkn).unwrap()),
224		};
225
226		TransferToken {
227			buff_tkn,
228			ctrl_desc,
229		}
230	}
231
232	// The descriptors returned by the iterator will be incomplete, as they do not
233	// have all the information necessary.
234	fn descriptor_iter(
235		buffer_tkn: &AvailBufferToken,
236	) -> Result<impl DoubleEndedIterator<Item = Self::Descriptor>, VirtqError> {
237		let send_desc_iter = buffer_tkn
238			.send_buff
239			.iter()
240			.map(|elem| (elem, elem.len(), virtq::DescF::empty()));
241		let recv_desc_iter = buffer_tkn
242			.recv_buff
243			.iter()
244			.map(|elem| (elem, elem.capacity(), virtq::DescF::WRITE));
245		let mut all_desc_iter =
246			send_desc_iter
247				.chain(recv_desc_iter)
248				.map(|(mem_descr, len, incomplete_flags)| {
249					let addr = mem_descr.addr().expose_provenance();
250					Self::Descriptor::incomplete_desc(
251						u64::try_from(addr).unwrap().into(),
252						len.into(),
253						incomplete_flags | virtq::DescF::NEXT,
254					)
255				});
256
257		let mut last_desc = all_desc_iter
258			.next_back()
259			.ok_or(VirtqError::BufferNotSpecified)?;
260		*last_desc.flags_mut() -= virtq::DescF::NEXT;
261
262		Ok(all_desc_iter.chain([last_desc]))
263	}
264}

308/// The struct represents buffers which are ready to be send via the
309/// virtqueue. Buffers can no longer be written or retrieved.
310pub struct TransferToken<Descriptor> {
311	/// Must be some in order to prevent drop
312	/// upon reuse.
313	buff_tkn: AvailBufferToken,
314	// Contains the [MemDescr] for the indirect table if the transfer is indirect.
315	ctrl_desc: Option<Box<[Descriptor]>>,
316}
317
318/// Public Interface for TransferToken
319impl<Descriptor> TransferToken<Descriptor> {
320	/// Returns the number of descritprors that will be placed in the queue.
321	/// This number can differ from the `BufferToken.num_descr()` function value
322	/// as indirect buffers only consume one descriptor in the queue, but can have
323	/// more descriptors that are accessible via the descriptor in the queue.
324	fn num_consuming_descr(&self) -> u16 {
325		if self.ctrl_desc.is_some() {
326			1
327		} else {
328			self.buff_tkn.num_descr()
329		}
330	}
331}

333#[derive(Debug)]
334pub enum BufferElem {
335	Sized(Box<dyn Any + Send, DeviceAlloc>),
336	Vector(Vec<u8, DeviceAlloc>),
337}
338
339impl BufferElem {
340	// Returns the initialized length of the element. Assumes [Self::Sized] to
341	// be initialized, since the type of the object is erased and we cannot
342	// detect if the content is actually a [MaybeUninit]. However, this function
343	// should be only relevant for read buffer elements, which should not be uninit.
344	// If the element belongs to a write buffer, it is likely that [Self::capacity]
345	// is more appropriate.
346	pub fn len(&self) -> u32 {
347		match self {
348			BufferElem::Sized(sized) => mem::size_of_val(sized.as_ref()),
349			BufferElem::Vector(vec) => vec.len(),
350		}
351		.try_into()
352		.unwrap()
353	}
354
355	pub fn capacity(&self) -> u32 {
356		match self {
357			BufferElem::Sized(sized) => mem::size_of_val(sized.as_ref()),
358			BufferElem::Vector(vec) => vec.capacity(),
359		}
360		.try_into()
361		.unwrap()
362	}
363
364	pub fn addr(&self) -> *const u8 {
365		match self {
366			BufferElem::Sized(sized) => ptr::from_ref(sized.as_ref()).cast::<u8>(),
367			BufferElem::Vector(vec) => vec.as_ptr(),
368		}
369	}
370}

372/// The struct represents buffers which are ready to be written or to be send.
373///
374/// BufferTokens can be written in two ways:
375/// * in one step via `BufferToken.write()
376///   * consumes BufferToken and returns a TransferToken
377/// * sequentially via `BufferToken.write_seq()
378///
379/// # Structure of the Token
380/// The token can potentially hold both a *send* and a *recv* buffer, but MUST hold
381/// one.
382/// The *send* buffer is the data the device will read during a transfer, the *recv* buffer
383/// is the data the device will write to during a transfer.
384///
385/// # What are Buffers
386/// A buffer represents multiple chunks of memory. Where each chunk can be of different size.
387/// The chunks are named descriptors in the following.
388///
389/// **For Example:**
390/// A buffer could consist of 3 descriptors:
391/// 1. First descriptor of 30 bytes
392/// 2. Second descriptor of 10 bytes
393/// 3. Third descriptor of 100 bytes
394///
395/// Each of these descriptors consumes one "element" of the
396/// respective virtqueue.
397/// The maximum number of descriptors per buffer is bounded by the size of the virtqueue.
398pub struct AvailBufferToken {
399	pub(crate) send_buff: SmallVec<[BufferElem; 2]>,
400	pub(crate) recv_buff: SmallVec<[BufferElem; 2]>,
401}
402
403pub(crate) struct UsedDeviceWritableBuffer {
404	elems: SmallVec<[BufferElem; 2]>,
405	remaining_written_len: u32,
406}
407
408impl UsedDeviceWritableBuffer {
409	pub fn pop_front_downcast<T>(&mut self) -> Option<Box<T, DeviceAlloc>>
410	where
411		T: Any,
412	{
413		if self.remaining_written_len < u32::try_from(size_of::<T>()).unwrap() {
414			return None;
415		}
416
417		// May panic, but we have written data remaining so there should always be an item
418		let elem = if self.elems.len() <= 2 {
419			self.elems.swap_remove(0)
420		} else {
421			self.elems.remove(0)
422		};
423
424		if let BufferElem::Sized(sized) = elem {
425			match sized.downcast::<MaybeUninit<T>>() {
426				Ok(cast) => {
427					self.remaining_written_len -= u32::try_from(size_of::<T>()).unwrap();
428					Some(unsafe { cast.assume_init() })
429				}
430				Err(_) => {
431					panic!("Attempted to downcast element to wrong type");
432				}
433			}
434		} else {
435			panic!("Attempted to pop elements in order different from insertion");
436		}
437	}
438
439	pub fn pop_front_vec(&mut self) -> Option<Vec<u8, DeviceAlloc>> {
440		if self.elems.is_empty() {
441			return None;
442		}
443
444		let elem = if self.elems.len() <= 2 {
445			self.elems.swap_remove(0)
446		} else {
447			self.elems.remove(0)
448		};
449
450		if let BufferElem::Vector(mut vector) = elem {
451			let new_len = u32::min(
452				vector.capacity().try_into().unwrap(),
453				self.remaining_written_len,
454			);
455			self.remaining_written_len -= new_len;
456			unsafe { vector.set_len(new_len.try_into().unwrap()) };
457			Some(vector)
458		} else {
459			panic!("Attempted to pop elements in order different from insertion");
460		}
461	}
462}

489// Public interface of BufferToken
490impl AvailBufferToken {
491	/// **Parameters**
492	/// * send: The slices that will make up the elements of the driver-writable buffer.
493	/// * recv: The slices that will make up the elements of the device-writable buffer.
494	///
495	/// **Reasons for Failure:**
496	/// * Both `send` and `recv` are empty, which is not allowed by Virtio.
497	///
498	/// * If one wants to have a structure in the style of:
499	/// ```
500	/// struct send_recv_struct {
501	///     // send_part: ...
502	///     // recv_part: ...
503	/// }
504	/// ```
505	/// they must split the structure after the send part and provide the respective part via the send argument and the respective other
506	/// part via the recv argument.
507	pub fn new(
508		send_buff: SmallVec<[BufferElem; 2]>,
509		recv_buff: SmallVec<[BufferElem; 2]>,
510	) -> Result<Self, VirtqError> {
511		if send_buff.is_empty() && recv_buff.is_empty() {
512			return Err(VirtqError::BufferNotSpecified);
513		}
514
515		Ok(Self {
516			send_buff,
517			recv_buff,
518		})
519	}
520}

522pub enum BufferType {
523	/// As many descriptors get consumed in the descriptor table as the sum of the numbers of slices in [AvailBufferToken::send_buff] and [AvailBufferToken::recv_buff].
524	Direct,
525	/// Results in one descriptor in the queue, hence consumes one element in the main descriptor table. The queue will merge the send and recv buffers as follows:
526	/// ```text
527	/// //+++++++++++++++++++++++
528	/// //+        Queue        +
529	/// //+++++++++++++++++++++++
530	/// //+ Indirect descriptor + -> refers to a descriptor list in the form of ->  ++++++++++++++++++++++++++
531	/// //+         ...         +                                                   +  Descriptors for send  +
532	/// //+++++++++++++++++++++++                                                   +  Descriptors for recv  +
533	/// //                                                                          ++++++++++++++++++++++++++
534	/// ```
535	/// As a result indirect descriptors result in a single descriptor consumption in the actual queue.
536	Indirect,
537}

539/// A newtype for descriptor ids, for better readability.
540#[derive(Clone, Copy)]
541struct MemDescrId(pub u16);
542
543/// MemPool allows to easily control, request and provide memory for Virtqueues.
544///
545/// The struct is initialized with a limit of free running "tracked"
546/// memory descriptor ids. As Virtqueus do only allow a limited amount of descriptors in their queue,
547/// the independent queues, can control the number of descriptors by this.
548struct MemPool {
549	pool: Vec<MemDescrId>,
550	limit: u16,
551}
552
553impl MemPool {
554	/// Returns a given id to the id pool
555	fn ret_id(&mut self, id: MemDescrId) {
556		self.pool.push(id);
557	}
558
559	/// Returns a new instance, with a pool of the specified size.
560	fn new(size: u16) -> MemPool {
561		MemPool {
562			pool: (0..size).map(MemDescrId).collect(),
563			limit: size,
564		}
565	}
566}

568/// Virtqeueus error module.
569///
570/// This module unifies errors provided to useres of a virtqueue, independent of the underlying

573	use crate::io;

575	#[derive(Debug)]
576	// Internal Error Handling for Buffers
577	pub enum BufferError {
578		WriteToLarge,
579		ToManyWrites,
580	}
581
582	// External Error Handling for users of the virtqueue.
583	pub enum VirtqError {
584		General,
585		/// Call to create a BufferToken or TransferToken without
586		/// any buffers to be inserted
587		BufferNotSpecified,
588		/// Selected queue does not exist or
589		/// is not known to the device and hence can not be used
590		QueueNotExisting(u16),
591		/// Signals, that the queue does not have any free descriptors
592		/// left.
593		/// Typically this means, that the driver either has to provide
594		/// "unsend" `TransferToken` to the queue (see Docs for details)
595		/// or the device needs to process available descriptors in the queue.
596		NoDescrAvail,
597		/// Indicates that a Bytes::new() call failed or generally that a buffer is to large to
598		/// be transferred as one. The Maximum size is u32::MAX. This also is the maximum for indirect
599		/// descriptors (both the one placed in the queue, as also the ones the indirect descriptor is
600		/// referring to).
601		BufferToLarge,
602		QueueSizeNotAllowed(u16),
603		FeatureNotSupported(virtio::F),
604		AllocationError,
605		IncompleteWrite,
606		NoNewUsed,
607	}
608
609	impl core::fmt::Debug for VirtqError {
610		fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
611			match self {
612				VirtqError::General => write!(f, "Virtq failure due to unknown reasons!"),
613				VirtqError::BufferNotSpecified => {
614					write!(f, "Virtq detected creation of Token, without a BuffSpec")
615				}
616				VirtqError::QueueNotExisting(_) => {
617					write!(f, "Virtq does not exist and can not be used!")
618				}
619				VirtqError::NoDescrAvail => write!(f, "Virtqs memory pool is exhausted!"),
620				VirtqError::BufferToLarge => {
621					write!(f, "Buffer to large for queue! u32::MAX exceeded.")
622				}
623				VirtqError::QueueSizeNotAllowed(_) => {
624					write!(f, "The requested queue size is not valid.")
625				}
626				VirtqError::FeatureNotSupported(_) => {
627					write!(f, "An unsupported feature was requested from the queue.")
628				}
629				VirtqError::AllocationError => write!(
630					f,
631					"An error was encountered during the allocation of the queue structures."
632				),
633				VirtqError::IncompleteWrite => {
634					write!(f, "A sized object was partially initialized.")
635				}
636				VirtqError::NoNewUsed => {
637					write!(f, "The queue does not contain any new used buffers.")
638				}
639			}
640		}
641	}
642
643	impl core::convert::From<VirtqError> for io::Error {
644		fn from(_: VirtqError) -> Self {
645			io::Error::EIO
646		}
