25/// Physical and virtual address range of the 2 MiB pages that map the kernel.
26static KERNEL_ADDR_RANGE: Lazy<Range<VirtAddr>> = Lazy::new(|| {
27	if cfg!(target_os = "none") {
28		// Calculate the start and end addresses of the 2 MiB page(s) that map the kernel.
29		env::get_base_address().align_down(LargePageSize::SIZE)
30			..(env::get_base_address() + env::get_image_size()).align_up(LargePageSize::SIZE)
31	} else {
32		VirtAddr::zero()..VirtAddr::zero()
33	}
34});
35
36pub(crate) fn kernel_start_address() -> VirtAddr {
37	KERNEL_ADDR_RANGE.start
38}
39
40pub(crate) fn kernel_end_address() -> VirtAddr {
41	KERNEL_ADDR_RANGE.end
42}

44#[cfg(target_os = "none")]
45pub(crate) fn init() {
46	use crate::arch::mm::paging;
47
48	Lazy::force(&KERNEL_ADDR_RANGE);
49
50	arch::mm::init();
51	arch::mm::init_page_tables();
52
53	let total_mem = physicalmem::total_memory_size();
54	let kernel_addr_range = KERNEL_ADDR_RANGE.clone();
55	info!("Total memory size: {} MiB", total_mem >> 20);
56	info!(
57		"Kernel region: {:p}..{:p}",
58		kernel_addr_range.start, kernel_addr_range.end
59	);
60
61	// we reserve physical memory for the required page tables
62	// In worst case, we use page size of BasePageSize::SIZE
63	let npages = total_mem / BasePageSize::SIZE as usize;
64	let npage_3tables = npages / (BasePageSize::SIZE as usize / mem::align_of::<usize>()) + 1;
65	let npage_2tables =
66		npage_3tables / (BasePageSize::SIZE as usize / mem::align_of::<usize>()) + 1;
67	let npage_1tables =
68		npage_2tables / (BasePageSize::SIZE as usize / mem::align_of::<usize>()) + 1;
69	let reserved_space = (npage_3tables + npage_2tables + npage_1tables)
70		* BasePageSize::SIZE as usize
71		+ 2 * LargePageSize::SIZE as usize;
72	#[cfg(any(target_arch = "x86_64", target_arch = "riscv64"))]
73	let has_1gib_pages = arch::processor::supports_1gib_pages();
74	let has_2mib_pages = arch::processor::supports_2mib_pages();
75
76	let min_mem = if env::is_uefi() {
77		// On UEFI, the given memory is guaranteed free memory and the kernel is located before the given memory
78		reserved_space
79	} else {
80		(kernel_addr_range.end.as_u64() - env::get_ram_address().as_u64() + reserved_space as u64)
81			as usize
82	};
83	info!("Minimum memory size: {} MiB", min_mem >> 20);
84	let avail_mem = total_mem
85		.checked_sub(min_mem)
86		.unwrap_or_else(|| panic!("Not enough memory available!"))
87		.align_down(LargePageSize::SIZE as usize);
88
89	let mut map_addr;
90	let mut map_size;
91	let heap_start_addr;
92
93	#[cfg(feature = "common-os")]
94	{
95		info!("Using HermitOS as common OS!");
96
97		// we reserve at least 75% of the memory for the user space
98		let reserve: usize = (avail_mem * 75) / 100;
99		// 64 MB is enough as kernel heap
100		let reserve = core::cmp::min(reserve, 0x0400_0000);
101
102		let virt_size: usize = reserve.align_down(LargePageSize::SIZE as usize);
103		let virt_addr =
104			self::virtualmem::allocate_aligned(virt_size, LargePageSize::SIZE as usize).unwrap();
105		heap_start_addr = virt_addr;
106
107		info!(
108			"Heap: size {} MB, start address {:p}",
109			virt_size >> 20,
110			virt_addr
111		);
112
113		#[cfg(any(target_arch = "x86_64", target_arch = "riscv64"))]
114		if has_1gib_pages && virt_size > HugePageSize::SIZE as usize {
115			// Mount large pages to the next huge page boundary
116			let npages = (virt_addr.align_up(HugePageSize::SIZE) - virt_addr) as usize
117				/ LargePageSize::SIZE as usize;
118			if let Err(n) = paging::map_heap::<LargePageSize>(virt_addr, npages) {
119				map_addr = virt_addr + n as u64 * LargePageSize::SIZE;
120				map_size = virt_size - (map_addr - virt_addr) as usize;
121			} else {
122				map_addr = virt_addr.align_up(HugePageSize::SIZE);
123				map_size = virt_size - (map_addr - virt_addr) as usize;
124			}
125		} else {
126			map_addr = virt_addr;
127			map_size = virt_size;
128		}
129
130		#[cfg(not(any(target_arch = "x86_64", target_arch = "riscv64")))]
131		{
132			map_addr = virt_addr;
133			map_size = virt_size;
134		}
135	}
136
137	#[cfg(not(feature = "common-os"))]
138	{
139		// we reserve 10% of the memory for stack allocations
140		let stack_reserve: usize = (avail_mem * 10) / 100;
141
142		// At first, we map only a small part into the heap.
143		// Afterwards, we already use the heap and map the rest into
144		// the virtual address space.
145
146		#[cfg(not(feature = "mmap"))]
147		let virt_size: usize = (avail_mem - stack_reserve).align_down(LargePageSize::SIZE as usize);
148		#[cfg(feature = "mmap")]
149		let virt_size: usize = ((avail_mem * 75) / 100).align_down(LargePageSize::SIZE as usize);
150
151		let virt_addr =
152			crate::mm::virtualmem::allocate_aligned(virt_size, LargePageSize::SIZE as usize)
153				.unwrap();
154		heap_start_addr = virt_addr;
155
156		info!(
157			"Heap: size {} MB, start address {:p}",
158			virt_size >> 20,
159			virt_addr
160		);
161
162		#[cfg(any(target_arch = "x86_64", target_arch = "riscv64"))]
163		if has_1gib_pages && virt_size > HugePageSize::SIZE as usize {
164			// Mount large pages to the next huge page boundary
165			let npages = (virt_addr.align_up(HugePageSize::SIZE) - virt_addr) / LargePageSize::SIZE;
166			if let Err(n) = paging::map_heap::<LargePageSize>(virt_addr, npages as usize) {
167				map_addr = virt_addr + n as u64 * LargePageSize::SIZE;
168				map_size = virt_size - (map_addr - virt_addr) as usize;
169			} else {
170				map_addr = virt_addr.align_up(HugePageSize::SIZE);
171				map_size = virt_size - (map_addr - virt_addr) as usize;
172			}
173		} else {
174			map_addr = virt_addr;
175			map_size = virt_size;
176		}
177
178		#[cfg(not(any(target_arch = "x86_64", target_arch = "riscv64")))]
179		{
180			map_addr = virt_addr;
181			map_size = virt_size;
182		}
183	}
184
185	#[cfg(any(target_arch = "x86_64", target_arch = "riscv64"))]
186	if has_1gib_pages
187		&& map_size > HugePageSize::SIZE as usize
188		&& map_addr.is_aligned_to(HugePageSize::SIZE)
189	{
190		let size = map_size.align_down(HugePageSize::SIZE as usize);
191		if let Err(num_pages) =
192			paging::map_heap::<HugePageSize>(map_addr, size / HugePageSize::SIZE as usize)
193		{
194			map_size -= num_pages * HugePageSize::SIZE as usize;
195			map_addr += num_pages as u64 * HugePageSize::SIZE;
196		} else {
197			map_size -= size;
198			map_addr += size;
199		}
200	}
201
202	if has_2mib_pages
203		&& map_size > LargePageSize::SIZE as usize
204		&& map_addr.is_aligned_to(LargePageSize::SIZE)
205	{
206		let size = map_size.align_down(LargePageSize::SIZE as usize);
207		if let Err(num_pages) =
208			paging::map_heap::<LargePageSize>(map_addr, size / LargePageSize::SIZE as usize)
209		{
210			map_size -= num_pages * LargePageSize::SIZE as usize;
211			map_addr += num_pages as u64 * LargePageSize::SIZE;
212		} else {
213			map_size -= size;
214			map_addr += size;
215		}
216	}
217
218	if map_size > BasePageSize::SIZE as usize && map_addr.is_aligned_to(BasePageSize::SIZE) {
219		let size = map_size.align_down(BasePageSize::SIZE as usize);
220		if let Err(num_pages) =
221			paging::map_heap::<BasePageSize>(map_addr, size / BasePageSize::SIZE as usize)
222		{
223			map_size -= num_pages * BasePageSize::SIZE as usize;
224			map_addr += num_pages as u64 * BasePageSize::SIZE;
225		} else {
226			map_size -= size;
227			map_addr += size;
228		}
229	}
230
231	let heap_end_addr = map_addr;
232
233	unsafe {
234		ALLOCATOR.init(
235			heap_start_addr.as_mut_ptr(),
236			(heap_end_addr - heap_start_addr) as usize,
237		);
238	}
239
240	info!("Heap is located at {heap_start_addr:p}..{heap_end_addr:p} ({map_size} Bytes unmapped)");
241}

248/// Maps a given physical address and size in virtual space and returns address.
249#[cfg(feature = "pci")]
250pub(crate) fn map(
251	physical_address: PhysAddr,
252	size: usize,
253	writable: bool,
254	no_execution: bool,
255	no_cache: bool,
256) -> VirtAddr {
257	use crate::arch::mm::paging::PageTableEntryFlags;
258	#[cfg(target_arch = "x86_64")]
259	use crate::arch::mm::paging::PageTableEntryFlagsExt;
260
261	let size = size.align_up(BasePageSize::SIZE as usize);
262	let count = size / BasePageSize::SIZE as usize;
263
264	let mut flags = PageTableEntryFlags::empty();
265	flags.normal();
266	if writable {
267		flags.writable();
268	}
269	if no_execution {
270		flags.execute_disable();
271	}
272	if no_cache {
273		flags.device();
274	}
275
276	let virtual_address = self::virtualmem::allocate(size).unwrap();
277	arch::mm::paging::map::<BasePageSize>(virtual_address, physical_address, count, flags);
278
279	virtual_address
280}

283/// unmaps virtual address, without 'freeing' physical memory it is mapped to!
284pub(crate) fn unmap(virtual_address: VirtAddr, size: usize) {
285	let size = size.align_up(BasePageSize::SIZE as usize);
286
287	if arch::mm::paging::virtual_to_physical(virtual_address).is_some() {
288		arch::mm::paging::unmap::<BasePageSize>(
289			virtual_address,
290			size / BasePageSize::SIZE as usize,
291		);
292		self::virtualmem::deallocate(virtual_address, size);
293	} else {
294		panic!(
295			"No page table entry for virtual address {:p}",
296			virtual_address
297		);
298	}
